from fastapi import FastAPI, HTTPException
from fastapi.responses import JSONResponse
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
import yt_dlp
import uvicorn
from typing import Optional

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

class RequestModel(BaseModel):
    url: str
    isAudioOnly: Optional[bool] = False

@app.post("/")
async def get_video(payload: RequestModel):
    try:
        url = payload.url
        is_audio = payload.isAudioOnly
        
        # Determine format
        if is_audio:
            format_str = 'bestaudio[ext=m4a]/bestaudio/best'
        else:
            format_str = 'b[ext=mp4]/best'

        ydl_opts = {
            'proxy': 'socks5://10.0.0.2:40000', # we will replace it later we are on host
            'extractor_args': {'youtube': {'player_client': ['android']}},
            'quiet': True,
            'no_warnings': True,
            'format': format_str,
        }
        # since we are running network: host, proxy is localhost
        ydl_opts['proxy'] = 'socks5://127.0.0.1:40000'

        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            info = ydl.extract_info(url, download=False)
            video_url = info.get('url')
            
            if not video_url and 'requested_formats' in info:
                video_url = info['requested_formats'][0].get('url')

            if not video_url:
                return JSONResponse(status_code=500, content={"status": "error", "error": {"code": "No stream extracted"}})

            return {
                "status": "stream",
                "url": video_url,
                "title": info.get("title", ""),
                "thumbnail": info.get("thumbnail", ""),
                "duration": info.get("duration", 0)
            }
    except Exception as e:
        return JSONResponse(status_code=500, content={"status": "error", "error": {"code": str(e)}})

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=9000)
EOF
